CREATE SEQUENCE SEQ_ID_EMPLEADO START WITH 1 INCREMENT BY 1;

CREATE OR REPLACE TRIGGER TRG_ID_EMPLEADO
BEFORE INSERT ON EMPLEADO
FOR EACH ROW
BEGIN
    IF :NEW.ID_EMPLEADO IS NULL THEN
        :NEW.ID_EMPLEADO := SEQ_ID_EMPLEADO.NEXTVAL;
    END IF;
END;
/

CREATE SEQUENCE SEQ_DIRECCION_ID START WITH 1 INCREMENT BY 1;

create or replace TRIGGER TRG_ID_DIRECCION
BEFORE INSERT ON DIRECCION
FOR EACH ROW
BEGIN
    -- Convertir el valor de la secuencia a texto antes de asignarlo
    IF :NEW.ID_DIRECCION IS NULL THEN
        :NEW.ID_DIRECCION := 'D'||LPAD(SEQ_DIRECCION_ID.NEXTVAL,2,'0');
    END IF;
END;
